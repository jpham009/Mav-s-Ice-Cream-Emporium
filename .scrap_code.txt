void Mainwin::on_restock_click(){
	Gtk::Dialog dialog{"Restock", *this};
	Gtk::HBox hbox_all;
	Gtk::HBox hb_containers;
	Gtk::VBox vb_containers_name;
	Gtk::VBox vb_containers_quantity;
	Gtk::VBox vb_containers_buttons;
	Gtk::HBox hb_scoops;
	Gtk::VBox vb_scoops_name;
	Gtk::VBox vb_scoops_quantity;
	Gtk::VBox vb_scoops_buttons;
	Gtk::HBox hb_toppings;
	Gtk::VBox vb_toppings_name;
	Gtk::VBox vb_toppings_quantity;
	Gtk::VBox vb_toppings_buttons;
	

	//vbox formatting
	hbox_all.set_spacing(30);
	hbox_all.set_margin_left(30);
	hbox_all.set_margin_right(30);
	hbox_all.set_margin_top(15);
	hbox_all.set_margin_bottom(15);
	
//containers
	hb_containers.set_halign(Gtk::ALIGN_START);
	hb_containers.set_valign(Gtk::ALIGN_START);
	hb_containers.set_margin_top(30);
	hb_containers.set_margin_bottom(30);
	
	vb_containers_name.set_halign(Gtk::ALIGN_START);
	vb_containers_name.set_valign(Gtk::ALIGN_START);
	vb_containers_name.set_margin_top(30);
	vb_containers_name.set_margin_bottom(30);
	
	vb_containers_quantity.set_halign(Gtk::ALIGN_CENTER);
	vb_containers_quantity.set_valign(Gtk::ALIGN_START);
	vb_containers_quantity.set_margin_top(30);
	vb_containers_quantity.set_margin_bottom(30);
	
	vb_containers_buttons.set_halign(Gtk::ALIGN_END);
	vb_containers_buttons.set_valign(Gtk::ALIGN_START);
	vb_containers_buttons.set_margin_top(30);
	vb_containers_buttons.set_margin_bottom(30);
//scoops
	hb_scoops.set_halign(Gtk::ALIGN_START);
	hb_scoops.set_valign(Gtk::ALIGN_START);
	hb_scoops.set_margin_top(30);
	hb_scoops.set_margin_bottom(30);
	
	vb_scoops_name.set_halign(Gtk::ALIGN_START);
	vb_scoops_name.set_valign(Gtk::ALIGN_START);
	vb_scoops_name.set_margin_top(30);
	vb_scoops_name.set_margin_bottom(30);
	
	vb_scoops_quantity.set_halign(Gtk::ALIGN_CENTER);
	vb_scoops_quantity.set_valign(Gtk::ALIGN_START);
	vb_scoops_quantity.set_margin_top(30);
	vb_scoops_quantity.set_margin_bottom(30);
	
	vb_scoops_buttons.set_halign(Gtk::ALIGN_END);
	vb_scoops_buttons.set_valign(Gtk::ALIGN_START);
	vb_scoops_buttons.set_margin_top(30);
	vb_scoops_buttons.set_margin_bottom(30);

//toppings
	hb_toppings.set_halign(Gtk::ALIGN_START);
	hb_toppings.set_valign(Gtk::ALIGN_START);
	hb_toppings.set_margin_top(30);
	hb_toppings.set_margin_bottom(30);
	
	vb_toppings_name.set_halign(Gtk::ALIGN_START);
	vb_toppings_name.set_valign(Gtk::ALIGN_START);
	vb_toppings_name.set_margin_top(30);
	vb_toppings_name.set_margin_bottom(30);
	
	vb_toppings_quantity.set_halign(Gtk::ALIGN_CENTER);
	vb_toppings_quantity.set_valign(Gtk::ALIGN_START);
	vb_toppings_quantity.set_margin_top(30);
	vb_toppings_quantity.set_margin_bottom(30);
	
	vb_toppings_buttons.set_halign(Gtk::ALIGN_CENTER);
	vb_toppings_buttons.set_valign(Gtk::ALIGN_START);
	vb_toppings_buttons.set_margin_top(30);
	vb_toppings_buttons.set_margin_bottom(30);

	

	dialog.add_button("Cancel", 0);
	dialog.add_button("OK", 1);
	
	std::string str_cont = " CONTAINERS";
	Gtk::Label l_container(str_cont);
	l_container.set_margin_bottom(10);
	l_container.set_margin_top(10);
	hb_containers.pack_start(l_container);
	for(Mice::Container container : _containers){
		Gtk::Label *container_name = Gtk::manage(new Gtk::Label(container.name()));
		Gtk::Label *container_quantity = Gtk::manage(new Gtk::Label(std::to_string(container.quantity())));  
		Gtk::Button *container_button = Gtk::manage(new Gtk::Button("Restock",1)); 
		vb_containers_name.pack_start(*container_name); 
		vb_containers_quantity.pack_start(*container_quantity);
		vb_containers_buttons.pack_start(*container_button);
	}
	
	Gtk::Label l_scoops(" SCOOPS");
	l_scoops.set_margin_bottom(10);
	l_scoops.set_margin_top(10);
	hb_scoops.pack_start(l_scoops);
		for(Mice::Scoop scoop : _scoops){
		Gtk::Label *scoop_name = Gtk::manage(new Gtk::Label(scoop.name()));
		Gtk::Label *scoop_quantity = Gtk::manage(new Gtk::Label(std::to_string(scoop.quantity())));  
		Gtk::Button *scoop_button = Gtk::manage(new Gtk::Button("Restock", 1)); 
		vb_scoops_name.pack_start(*scoop_name); 
		vb_scoops_quantity.pack_start(*scoop_quantity);
		vb_scoops_buttons.pack_start(*scoop_button);
		}
	Gtk::Label l_toppings(" TOPPINGS");
l_toppings.set_margin_bottom(10);
	l_toppings.set_margin_top(10);
	hb_toppings.pack_start(l_toppings);
	for(Mice::Topping topping : _toppings){
		Gtk::Label *topping_name = Gtk::manage(new Gtk::Label(topping.name()));
		Gtk::Label *topping_quantity = Gtk::manage(new Gtk::Label(std::to_string(topping.quantity())));  
		Gtk::Button *topping_button = Gtk::manage(new Gtk::Button("Restock",1)); 
		vb_toppings_name.pack_start(*topping_name); 
		vb_toppings_quantity.pack_start(*topping_quantity);
		vb_toppings_buttons.pack_start(*topping_button);
		}
	
	hb_containers.pack_start(vb_containers_name);
	hb_containers.pack_start(vb_containers_quantity);
	hb_containers.pack_start(vb_containers_buttons);
	hb_scoops.pack_start(vb_scoops_name);
	hb_scoops.pack_start(vb_scoops_quantity);
	hb_scoops.pack_start(vb_scoops_buttons);
	hb_toppings.pack_start(vb_toppings_name);
	hb_toppings.pack_start(vb_toppings_quantity);
	hb_toppings.pack_start(vb_toppings_buttons);

	hbox_all.pack_start(hb_containers);
	hbox_all.pack_start(hb_scoops);
	hbox_all.pack_start(hb_toppings);
	dialog.get_vbox()->pack_start(hbox_all);
	dialog.show_all();
	dialog.run();
